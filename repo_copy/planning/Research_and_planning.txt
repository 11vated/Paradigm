No API versioning — Breaking changes to /api/seeds structure would break all clients.
High Priority
Seed lineage tracking — $lineage field exists but mutation/crossover operators don't update it properly for complex multi-step operations.
Gene validation incomplete — GSPL compiler doesn't catch all invalid gene combinations. E.g., character with negative strength validates when it shouldn't.
QFT solver stability — FDTD has implicit stability requirements (CFL condition: dt ≤ dx/c). Code doesn't verify these; silently wrong if violated.
Medium Priority
No audit log for evolution operations — Seeds are created but the "why" (which agents bred them, under what fitness function) isn't tracked beyond $lineage.
Cache invalidation — Redis cache key structure is ad-hoc; no coherence guarantees if multiple processes modify same seed.
No GraphQL — REST API is RESTful but verbose; GraphQL would reduce bandwidth for large seed trees.

// src/lib/agent/memory.ts
export class AgentMemory {
  private entries: MemoryEntry[] = [];
  private maxEntries: number;
  private turnCounter: number = 0;
  constructor(maxEntries: number = 25) {
    this.maxEntries = Math.max(1, Math.min(50, maxEntries));
  }
  addUserMessage(content: string, seedsReferenced: string[] = []): void {
    this.turnCounter++;
    this.entries.push({
      turn: this.turnCounter,
      role: 'user',
      content,
      seedsReferenced,
      seedsCreated: [],
      intent: 'unknown',
      timestamp: Date.now(),
    });
    this.trim();
  }
  addAgentResponse(
    content: string,
    intent: AgentIntent,
    seedsCreated: string[] = [],
    seedsReferenced: string[] = [],
  ): void {
    this.entries.push({
      turn: this.turnCounter,
      role: 'agent',
      content,
      seedsReferenced,
      seedsCreated,
      intent,
      timestamp: Date.now(),
    });
    this.trim();
  }
  private trim(): void {
    if (this.entries.length > this.maxEntries * 2) {
      this.entries = this.entries.slice(-this.maxEntries * 2);
    }
  }
  getContext(maxTurns?: number): MemoryEntry[] {
    const limit = maxTurns ?? this.maxEntries;
    return this.entries.slice(-limit * 2);
  }
  formatForPrompt(maxTokensEstimate: number = 1000): string {
    const entries = this.getContext();
    if (entries.length === 0) return '';
    const lines: string[] = [];
    let charCount = 0;
    const charLimit = maxTokensEstimate * 4;
    for (let i = entries.length - 1; i >= 0; i--) {
      const e = entries[i];
      const line = [${e.role}] ${e.content};
      if (charCount + line.length > charLimit) break;
      lines.unshift(line);
      charCount += line.length;
    }
    return lines.join('\n');
  }
  get turnCount(): number { return this.turnCounter; }
  get size(): number { return this.entries.length; }
  setMaxEntries(n: number): void { this.maxEntries = Math.max(1, Math.min(50, n)); }
}

const INTENT_TIER: Record<AgentIntent, InferenceTier> = {
  'create_seed': InferenceTier.FAST,
  'mutate_seed': InferenceTier.ROUTER,
  'breed_seeds': InferenceTier.FAST,
  'compose_seed': InferenceTier.ROUTER,
  'grow_seed': InferenceTier.KERNEL,
  'evolve_seeds': InferenceTier.FAST,
  'describe_domain': InferenceTier.KERNEL,
  'list_domains': InferenceTier.KERNEL,
  'help': InferenceTier.KERNEL,
  'unknown': InferenceTier.KERNEL,
};
export function parseQuery(query: string): ParsedQuery {
  const lower = query.toLowerCase();
  const words = new Set(lower.split(/\s+/));
  let intent: AgentIntent = 'unknown';
  let confidence = 0.0;
  // ── Intent Patterns ─────────────────────────────────────────────────────
  const patterns: [AgentIntent, RegExp, number][] = [
    ('help', /^(help|what can you|how do|guide|info|list operations)/, 1.0),
    ('list_domains', /^list (all )?(domains|engines|environments)/, 1.0),
    ('describe_domain', /^(describe|tell me about|explain) (the )?([a-z_]+) domain/, 0.9),
    ('create_seed', /(create|make|generate|new|spawn|birth).*seed/, 0.85),
    ('mutate_seed', /(mutate|tweak|alter|modify).*seed/, 0.85),
    ('breed_seeds', /(breed|cross|mate|combine).*seeds/, 0.9),
    ('evolve_seeds', /(evolve|develop|improve|enhance).*seeds?/, 0.8),
    ('compose_seed', /(compose|transform|convert|map).*seed.*to/, 0.85),
    ('grow_seed', /(grow|develop|cultivate|expand).*seed/, 0.8),
  ];
  for (const [pat_intent, regex, weight] of patterns) {
    if (regex.test(lower)) {
      intent = pat_intent;
      confidence = weight;
      break;
    }
  }
  // ── Entity Extraction ───────────────────────────────────────────────────
  const entities: Partial<ParsedQuery['entities']> = {};
  // Domain extraction
  const domainMatch = query.match(/(?:in|to|from|domain:?)\s+([a-z_]+)/i);
  if (domainMatch) entities.domain = domainMatch[1];
  // Gene type extraction
  const geneMatch = query.match(/(?:gene|type:?)\s+([a-z_]+)/i);
  if (geneMatch) entities.geneType = geneMatch[1];
  // Numeric parameters
  const rateMatch = query.match(/rate[\s:=]+([0-9.]+)/i);
  if (rateMatch) entities.mutationRate = Math.min(1, Math.max(0, parseFloat(rateMatch[1])));
  const popMatch = query.match(/(?:population|pop)[\s:=]+(\d+)/i);
  if (popMatch) entities.populationSize = Math.min(100, parseInt(popMatch[1]));
  return {
    intent,
    tier: INTENT_TIER[intent],
    entities: entities as ParsedQuery['entities'],
    confidence,
    rawInput: query,
  };
}
export function buildPlan(parsed: ParsedQuery, seeds: any[]): ReasoningPlan {
  const steps: PlanStep[] = [];
  switch (parsed.intent) {
    case 'create_seed':
      steps.push({
        operation: 'create_seed',
        params: { domain: parsed.entities.domain || 'character', name: New ${parsed.entities.domain || 'Seed'} },
        status: 'pending',
      });
      break;
    case 'mutate_seed':
      if (seeds.length > 0) {
        steps.push({
          operation: 'mutate_seed',
          params: { seedIndex: -1, rate: parsed.entities.mutationRate || 0.15 },
          status: 'pending',
        });
      }
      break;
    case 'breed_seeds':
      if (seeds.length >= 2) {
        steps.push({
          operation: 'breed_seeds',
          params: { indexA: -2, indexB: -1 },
          status: 'pending',
        });
      }
      break;
    // ... additional intents
  }
  return {
    intent: parsed.intent,
    steps,
    status: 'pending',
    currentStep: 0,
  };
}
export async function executePlan(
  plan: ReasoningPlan,
  context: ToolContext,
): Promise<ReasoningPlan> {
  const mutableSeeds = [...context.seeds];
  for (let i = 0; i < plan.steps.length; i++) {
    const step = plan.steps[i];
    plan.currentStep = i;
    step.status = 'running';
    const tool = AGENT_TOOLS.get(step.operation);
    if (tool) {
      try {
        const toolCtx: ToolContext = { ...context, seeds: mutableSeeds };
        const result = await tool.execute(step.params, toolCtx);
        step.status = 'completed';
        step.result = result;
        if (result?.seedsCreated) mutableSeeds.push(...result.seedsCreated);
      } catch (e: any) {
        step.status = 'failed';
        step.error = e.message;
        plan.status = 'failed';
        break;
      }
    }
  }
  if (!plan.steps.some(s => s.status === 'failed')) {
    plan.status = 'completed';
  }
  return plan;
}
export function buildResponse(
  plan: ReasoningPlan,
  parsed: ParsedQuery,
  startTime: number,
): AgentResponse {
  const timing = { parseMs: 0, planMs: 0, executeMs: Date.now() - startTime, totalMs: Date.now() - startTime };
  if (plan.status === 'failed') {
    return {
      success: false,
      intent: plan.intent,
      tier: InferenceTier.KERNEL,
      message: ${plan.intent} operation failed at step ${plan.currentStep + 1},
      data: { error: plan.steps[plan.currentStep]?.error },
      timing,
    };
  }
  const createdSeeds: any[] = [];
  const messages: string[] = [];
  for (const step of plan.steps) {
    if (step.result?.seedsCreated) createdSeeds.push(...step.result.seedsCreated);
    if (step.result?.message) messages.push(step.result.message);
  }
  return {
    success: true,
    intent: plan.intent,
    tier: InferenceTier.KERNEL,
    message: messages.join(' → ') || Executed ${plan.intent} successfully.,
    data: { seeds: createdSeeds },
    timing,
  };
}

  context: ToolContext,

export enum InferenceTier { KERNEL = 0, ROUTER = 1, FAST = 2, DEEP = 3 }
export interface InferenceResponse {
  text: string;
  model: string;
  tier: InferenceTier;
  tokensUsed: number;
}
export class LocalInferenceClient {
  private serverUrl: string | null;
  private maxTier: InferenceTier;
  constructor() {
    this.serverUrl = process.env.LLAMA_SERVER_URL || null;
    this.maxTier = this.detectMaxTier();
  }
  private detectMaxTier(): InferenceTier {
    if (!this.serverUrl) return InferenceTier.KERNEL;
    // Would probe /health endpoint in full implementation
    return InferenceTier.FAST;
  }
  maxAvailableTier(): InferenceTier {
    return this.maxTier;
  }
  async generate(
    request: { prompt: string; systemPrompt?: string; maxTokens?: number; temperature?: number },
    preferredTier: InferenceTier = InferenceTier.FAST,
  ): Promise<InferenceResponse> {
    const tier = Math.min(preferredTier, this.maxTier);
    if (tier === InferenceTier.KERNEL) {
      // No inference available — return empty response
      return { text: '', model: 'kernel', tier, tokensUsed: 0 };
    }
    if (!this.serverUrl) {
      return { text: '', model: 'offline', tier: InferenceTier.KERNEL, tokensUsed: 0 };
    }
    try {
      const response = await fetch(${this.serverUrl}/completion, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          prompt: request.prompt,
          n_predict: request.maxTokens || 256,
          temperature: request.temperature || 0.7,
          system_prompt: request.systemPrompt,
        }),
      });
      if (!response.ok) throw new Error(Inference server error: ${response.status});
      const data = await response.json();
      return {
        text: data.content || '',
        model: data.model || 'unknown',
        tier,
        tokensUsed: data.tokens_predicted || 0,
      };
    } catch (e) {
      console.warn('[Agent] Inference error, falling back to kernel:', e);
      return { text: '', model: 'error', tier: InferenceTier.KERNEL, tokensUsed: 0 };
    }
  }
}
let globalClient: LocalInferenceClient;
export function getInferenceClient(): LocalInferenceClient {
  if (!globalClient) globalClient = new LocalInferenceClient();
  return globalClient;
}

        }
        break;

    const detectionResults = await this.automatedDetection(seed);
    if (detectionResults.highConfidenceMatch) {

// IMPLEMENTATION: Authenticity Calculation Engine
class AuthenticityEngine {
  async calculateAuthenticityScore(seed: Seed): Promise<number> {
    // Step 1: Gather metrics
    const intentionality = await this.scoreIntentionality(seed);
    const decisionMaking = await this.scoreDecisionMaking(seed);
    const uniqueness = await this.scoreUniqueness(seed);
    const timeInvested = await this.scoreTimeInvested(seed);
    
    // Step 2: Weight by importance
    const weights = {
      intentionality: 0.4,    // Most important: did creator have clear vision?
      decisionMaking: 0.35,   // Did creator make meaningful choices?
      uniqueness: 0.15,       // Is this new/original?
      timeInvested: 0.1,      // Did they put in effort?
    };
    
    const overall = (
      intentionality * weights.intentionality +
      decisionMaking * weights.decisionMaking +
      uniqueness * weights.uniqueness +
      timeInvested * weights.timeInvested
    );
    
    return overall;
  }
  
  async scoreIntentionality(seed: Seed): Promise<number> {
    // Did the creator have a clear vision?
    // Score based on:
    // ├─ Consistency of choices (all choices point same direction?)
    // ├─ Explanation quality (creator described what they wanted?)
    // └─ Vision clarity (can you infer intent from final seed?)
    
    const visionPrompt = seed.creationHistory.initialPrompt;
    const choices = seed.creationHistory.humanChoices;
    
    // Do all choices align with original prompt?
    const alignment = await this.calculateAlignmentWithVision(choices, visionPrompt);
    
    // How specific was the prompt?
    const specificity = await this.calculatePromptSpecificity(visionPrompt);
    
    return (alignment * 0.7 + specificity * 0.3);
  }
  
  async scoreDecisionMaking(seed: Seed): Promise<number> {
    // How many meaningful decisions did creator make?
    // Score based on:
    // ├─ Number of decision points (more = more authentic)
    // ├─ Uniqueness of choices (did they pick the obvious option?)
    // └─ Coherence (did choices make sense together?)
    
    const choices = seed.creationHistory.humanChoices;
    
    if (choices.length === 0) return 0.1;  // No decisions = no authenticity
    if (choices.length === 1) return 0.3;  // Just one click = low
    if (choices.length > 10) return 0.95;  // Lots of iteration = high
    
    const decisionCount = Math.min(choices.length, 10) / 10;
    
    // Are choices the "obvious" ones or did creator diverge?
    const uniqueness = await this.calculateChoiceUniqueness(choices);
    
    // Do choices make coherent sense?
    const coherence = await this.calculateChoiceCoherence(choices);
    
    return (decisionCount * 0.5 + uniqueness * 0.3 + coherence * 0.2);
  }
  
  async scoreUniqueness(seed: Seed): Promise<number> {
    // Is this seed genetically unique?
    // Compare to all historical seeds
    
    const similarSeeds = await this.findSimilarSeeds(seed);
    
    if (similarSeeds.length === 0) return 1.0;  // Totally unique
    
    const avgSimilarity = similarSeeds.reduce((a, b) => a + b.similarity, 0) / similarSeeds.length;
    
    // 0.0 similarity = 1.0 uniqueness score
    // 0.5 similarity = 0.5 uniqueness score
    // 1.0 similarity = 0.0 uniqueness score
    
    return 1.0 - avgSimilarity;
  }
  
  async scoreTimeInvested(seed: Seed): Promise<number> {
    // Did creator invest real time?
    
    const humanMinutes = seed.creationHistory.humanTimeInvested;
    const computeSeconds = seed.creationHistory.computeTimeInvested;
    
    // Ideal: Human time >> Compute time
    // Ratio of 22.5:1 is good
    
    const ratio = humanMinutes * 60 / computeSeconds;
    
    if (ratio < 5) return 0.2;      // Mostly compute, little human
    if (ratio < 10) return 0.5;     // Balanced
    if (ratio < 30) return 0.9;     // Mostly human
    return 1.0;                     // Very human-intensive
  }
  
  displayAuthenticityLabel(score: number): string {
    // Convert score to market-friendly label
    
    if (score > 0.9) return "100% Human-Crafted";
    if (score > 0.75) return "Authentically Crafted";
    if (score > 0.5) return "AI-Assisted Creation";
    return "Algorithmic Synthesis";
  }
  
  calculatePriceMultiplier(score: number): number {
    // Authenticity affects market price
    
    if (score > 0.9) return 3.0;   // 3x premium
    if (score > 0.75) return 1.5;  // 50% premium
    if (score > 0.5) return 1.0;   // Baseline
    return 0.3;                    // 70% discount
  }
}
